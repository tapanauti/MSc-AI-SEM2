# Sigmoid Function Definition


def sg(z):
    return 1 / (1 + np.exp(-z))


# 1 Author: Tapan
# Calculate loss function
def loss_gradient(theta, x, y):
    z = np.dot(x, theta)
    h = sg(z)
    loss = (-y * np.log(h) - (1 - y) * np.log(1 - h)).mean()
    gdn = np.dot(x.T, (y - h)) / y.size

    # 1 Author: Atharva
    # Normalising Gradient Descent
    grad = gdn + 300 / (2 * y.size) * np.concatenate(([0], theta[1:])).T
    return loss, grad


# 1 Author: Tapan
# Training Function
def model_fit(x, y, total_iterations=5000):
    final_theta = []
    label = np.unique(y)
    all_loss = np.zeros(total_iterations)

    for c in label:
        Y = np.where(y == c, 1, 0)
        theta = np.zeros(x.shape[1])
        for i in range(total_iterations):
            all_loss[i], grad = loss_gradient(theta, x, Y)
            theta += 0.01 * grad
        final_theta.append(theta)
    return final_theta, label, all_loss


# 1 Author: Atharva
# Predict Function
def predict_prob(label, theta, x):
    preds = [np.argmax([sg(i@j) for j in final_theta]) for i in x]
    return [label[p] for p in preds]


# 1 Author: Atharva
